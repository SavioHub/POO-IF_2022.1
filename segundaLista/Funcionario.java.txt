	public Funcionario() {

		this.setor = "desenvolvedor por exemplo";
		this.cpf = "093.385.088-45";

	public void setNome(String nome) {

	public void valorSalarial(String senioridadeDev) {
		
		if(senioridadeDev == "Junior") {
			this.salarioMinimo = 2000.00;
			System.out.println("Você é um dev junior!");
			System.out.println("Seu sário é de no minimo "+this.salarioMinimo);
			
		}else if(senioridadeDev == "Pleno") {

		}else if(senioridadeDev == "Senior") {
			this.salarioMinimo = 4500.00;

			System.out.println("Seu sário é de no minimo "+this.salarioMinimo);
		}else {
			System.out.println("Algo de errado, tente novamente");
		}
	}
